#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Student {
    int id;
    char name[50];
    int age;
    float marks;
};

void addStudent(struct Student s[], int *n) {
    printf("\nEnter Student ID: ");
    scanf("%d", &s[*n].id);
    printf("Enter Name: ");
    scanf(" %[^\n]", s[*n].name);
    printf("Enter Age: ");
    scanf("%d", &s[*n].age);
    printf("Enter Marks: ");
    scanf("%f", &s[*n].marks);
    (*n)++;
    printf("\n✅ Student added successfully!\n");
}

void displayStudents(struct Student s[], int n) {
    if (n == 0) {
        printf("\nNo students to display!\n");
        return;
    }
    printf("\n------ Student Records ------\n");
    for (int i = 0; i < n; i++) {
        printf("ID: %d | Name: %s | Age: %d | Marks: %.2f\n", s[i].id, s[i].name, s[i].age, s[i].marks);
    }
}

void searchStudent(struct Student s[], int n) {
    int id, found = 0;
    printf("\nEnter ID to search: ");
    scanf("%d", &id);
    for (int i = 0; i < n; i++) {
        if (s[i].id == id) {
            printf("\nStudent Found!\n");
            printf("ID: %d | Name: %s | Age: %d | Marks: %.2f\n", s[i].id, s[i].name, s[i].age, s[i].marks);
            found = 1;
            break;
        }
    }
    if (!found)
        printf("\nStudent not found!\n");
}

void deleteStudent(struct Student s[], int *n) {
    int id, found = 0;
    printf("\nEnter ID to delete: ");
    scanf("%d", &id);
    for (int i = 0; i < *n; i++) {
        if (s[i].id == id) {
            for (int j = i; j < *n - 1; j++) {
                s[j] = s[j + 1];
            }
            (*n)--;
            found = 1;
            printf("\n✅ Student deleted successfully!\n");
            break;
        }
    }
    if (!found)
        printf("\nStudent not found!\n");
}

int main() {
    struct Student s[100];
    int n = 0, choice;

    while (1) {
        printf("\n====== Student Management System ======\n");
        printf("1. Add Student\n");
        printf("2. Display Students\n");
        printf("3. Search Student\n");
        printf("4. Delete Student\n");
        printf("5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1: addStudent(s, &n); break;
            case 2: displayStudents(s, n); break;
            case 3: searchStudent(s, n); break;
            case 4: deleteStudent(s, &n); break;
            case 5: exit(0);
            default: printf("\n❌ Invalid choice! Try again.\n");
        }
    }
    return 0;
}
